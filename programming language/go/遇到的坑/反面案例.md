## 案例一 goroutine 和channel的使用

读者可以仔细阅读下面的函数，看看是否存在问题。若存在问题是哪些问题，又该如何解决

~~~go
func (l *LogForwardING) ForwardIng(ctx *gin.Context) {
 mps := make([]map[string]interface{}, 0)
 if err := ctx.ShouldBindJSON(&mps); err != nil {
  logger.Logger.Errorf("forwardIng params faild: %v", err)
  ctx.JSON(http.StatusOK,
   res.ReturnError(e.InvalidParams, e.GetMsg(e.InvalidParams)))
  return
 }
 logger.Logger.Infof("data success received %v entry...", len(mps))
 //创建数据缓冲区
 jobs := make(chan *logForwardChan, forwardingGroutineMax)
 //创建处理完成的数据接收缓冲区
 dataJobs := make(chan interface{}, len(mps))
 //开启forwardingGroutineMax 个协程，监听jobs数据
 for k := 1; k < forwardingGroutineMax; k++ {
  go func() {
   for job := range jobs {
    dataJobs <- l.runParamsParsing(job.TagName, job.Data)
   }
  }()
 }

 //将收到的日志数据解析，写入到数据缓冲区jobs
 for _, mp := range mps {
  tag, ok := mp["tag-name"]
  if !ok {
   logger.Logger.Errorf("tag not in params: %v", mp)
   continue
  }

  mpByte, err := json.Marshal(mp)
  if err != nil {
   logger.Logger.Errorf("mp marshal byte faild: %v", err)
   continue
  }
  jobs <- &logForwardChan{TagName: tag.(string), Data: mpByte}

 }

 //创建数据存储列表，将处理完成的数据，写入数据存储列表
 saveDatas := make([]interface{}, 0)
 for k := 0; k < len(mps); k++ {
  saveData := <-dataJobs
  if saveData == nil {
   break
  }
  saveDatas = append(saveDatas, saveData)
 }

 //调用模型数据存储
 l.LogForwardIngEr.SaveData(saveDatas)
 close(jobs)
 close(dataJobs)
 return
}
~~~
