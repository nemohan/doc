## invalid indirect read from stack

写ebpf程序的时候，遇到了"invalid indirect read from stack"这个问题，详细信息如下：

~~~
panic: program sys_enter_sendmmsg: load program: permission denied: invalid indirect read from stack off -80+12 size 16 (529 line(s) omitted)
~~~

从字面意思看是间接读取某个栈的位置无效，给出的错误信息只包含是在哪个函数触发的，没有具体到是哪一行代码。首先想到如何确定是哪一行代码导致了这个问题，以及导致这个问题的原因。

确定是哪一行代码有两种思路:

* 比较笨的方法，通过注释代码的方法确定
* 反汇编 llvm-objdump。

通过反汇编的方法并没有找到使用"-80 + 12"的代码片段，所以就使用了方法1，发现是调用bpf_map_update_elem时触发的问题。

为什么bpf_map_update_elem会导致这个问题呢，开始走了不少弯路，直到发现了https://www.kernel.org/doc/html/v5.17/bpf/verifier.html这篇文档，上面记录了一些ebpf verifier校验不通过及原因的代码样例，其中就有"invalid indirect read from stack"的错误样例。结合样例和代码的反汇编大致确定了问题所在

导致上述问题的原因:

结构体write_msg_ctx 实际占用了16字节的栈空间(内存对齐)，在栈上的起始地址是r10 - 80(结束地址是r10-64)。因为write_msg_ctx.len是int类型（4字节)，给其赋值时 的"63 1a b8 ff 00 00 00 00 *(u32 *)(r10 - 72) = r1 " 代码只初始化了r10 - 72 开始的4字节，r10-68 开始的4字节并未初始化(即报错信息中的-80 + 12的位置)，所以在访问r10-68这块空间时就会出现"invalid indirect read from stack"

程序片段

~~~c
struct write_msg_ctx{
    void *data;
    int len;
};

SEC("tracepoint/syscalls/sys_enter_sendmmsg")
int static inline sys_enter_sendmmsg(struct enter_sendmmsg_arg *ctx){
    if(!is_valid_pid()){
        return 0;
    }
   struct mmsghdr *hdr = ctx->msgvec;
    struct write_msg_ctx write_msg = {
        .data = hdr,
        .len = ctx->vlen ,
    };
    handle_socket_write_enter( ctx, ctx->fd, &write_msg);
   return 0;
}


static int  __always_inline handle_socket_write_enter( void *ctx, int fd, void *key_value){
    //.......
    struct write_arg_key_v1 key = {
        .tgid_pid = tgid_pid,
    };
    int ret = bpf_map_update_elem(&write_msgs_map, &key, key_value, BPF_ANY);
    if(ret < 0){
        bpf_printk("map_update_elem failed. %s ret:%d\n", comm, ret);
    }
    //.............
}
~~~

## 循环的使用

拦截"sys_exit_sendmmsg"系统调用，捕获其第二个struct mmsghdr的数组类型参数时，因为是个数组，需要用到循环对其进行处理。使用循环时遇到了"argument list too long"的错误。

报错信息如下

~~~
panic: program sys_exit_sendmmsg: load program: argument list too long: 91: (18) r1 = 0x253a6d6d6f632067: 93: (truncated, 1496 line(s) omitted)
~~~



sendmmsg函数原型

~~~c
int sendmmsg(int sockfd, struct mmsghdr *msgvec, unsigned int vlen,
                    int flags);
~~~

在网上找了一些资料，有一些使用循环的建议:

* 使用 #pragma unroll 展开循环
* 使用bpf_loop函数

第一次尝试了#pragma unroll并没有解决问题。之前使用#pragma unroll时，控制循环结束的条件都是编译期已知的常量。

~~~c
    struct user_msghdr umsg_hdr ;
    int64_t len = msg_ctx->len;
    struct mmsghdr msghdr;
	#pragma unroll
    for(int i = 0; i < len; i++){
        bpf_probe_read_user(&msghdr, sizeof(msghdr), &(hdr[i]));
        bpf_printk("sys_exit_sendmmsg comm:%s len:%d\n", comm, msghdr.msg_len);
    }
~~~



现在控制循环结束的条件存储在一个变量中，估计ebpf verifier不支持，所以又尝试了明确指定结束条件超过某个编译期常量时，使用该常量作为结束条件，代码片段如下，仍然没有解决问题。

~~~c
    struct user_msghdr umsg_hdr ;
    int64_t len = msg_ctx->len;
    if(len > 2){
        len = 2;
    }
    struct mmsghdr msghdr;
	#pragma unroll
    for(int i = 0; i < len; i++){
        bpf_probe_read_user(&msghdr, sizeof(msghdr), &(hdr[i]));
        bpf_printk("sys_exit_sendmmsg comm:%s len:%d\n", comm, msghdr.msg_len);
    }
~~~

看到这个网页 https://ebpf-docs.dylanreimerink.nl/linux/concepts/loops/ 上关于"Bounded loops"的介绍时，尝试将控制结束条件的变量的类型类型由int64_t改为int8_t，成功解决问题

~~~c
    struct user_msghdr umsg_hdr ;
    int8_t len = msg_ctx->len;
    if(len > 2){
        len = 2;
    }
    struct mmsghdr msghdr;
	#pragma unroll
    for(int i = 0; i < len; i++){
        bpf_probe_read_user(&msghdr, sizeof(msghdr), &(hdr[i]));
        bpf_printk("sys_exit_sendmmsg comm:%s len:%d\n", comm, msghdr.msg_len);
    }
~~~



使用循环的代码片段:

~~~c
SEC("tracepoint/syscalls/sys_exit_sendmmsg")
static int __always_inline sys_exit_sendmmsg(struct exit_sendmmsg_arg *ctx){
    if(!is_valid_pid()){
        return 0;
    }
    struct write_arg_key_v1 key = {
        .tgid_pid = bpf_get_current_pid_tgid(),
        //.fd = 0,
    };
    char comm[TASK_COMM_LEN];
    bpf_get_current_comm(comm, sizeof(comm));
    struct write_msg_ctx *msg_ctx = bpf_map_lookup_elem(&write_msgs_map, &key);
    if(!msg_ctx){
        bpf_printk("sys_exit_sendmmsg no value found for comm:%s\n", comm);
        return 0;
    }
    struct mmsghdr *hdr = msg_ctx->data; 
    if(!hdr || msg_ctx->len <=0){
        return 0;
    }
    bpf_printk("exit hdr:%llx %lld\n",hdr ,msg_ctx->len);
    struct user_msghdr umsg_hdr ;
    struct iovec iovec; 
    //len类型最初为int64_t，虽然明确指定了数组长度最大为2，但仍有上述错误
    int8_t len = msg_ctx->len;
    if(len > 2){
        len = 2;
    }
    struct mmsghdr msghdr;

    for(int i = 0; i < len; i++){
        bpf_probe_read_user(&msghdr, sizeof(msghdr), &(hdr[i]));
        bpf_printk("sys_exit_sendmmsg comm:%s len:%d\n", comm, msghdr.msg_len);
    }
~~~

## bad map relo against '.L.str.28' in section '.rodata.str1.1'

os: 5.4.0-193-generic

## 入侵go语言程序

入侵go语言程序中的函数时，使用了BPF_MAP_TYPE_PERCPU_HASH类型的map保存函数入参，以进程id+协程id作为key。当入侵函数返回点时，再查找保存的信息，查到了无效的信息。因为协程可能被调度到其他线程执行，保存数据的线程和查找数据时的线程不是同一个，且运行在不同的cpu上，从而导致查到无效数据
